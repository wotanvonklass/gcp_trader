# Pako Web - News Trading Monitor Specification

## Overview

Real-time monitoring dashboard for the news trading pipeline. Watch news arrive, see trading decisions made, monitor active strategies, and analyze performance.

**Primary Focus:** Monitor the news â†’ decision â†’ strategy â†’ trade pipeline in real-time.

**Tech Stack:** React + TypeScript + Vite + TailwindCSS + Lightweight Charts

**Event Flow:**
1. Scraper publishes news to Pub/Sub
2. News Trader receives news, makes decisions, spawns strategies
3. News Trader POSTs events to News API at each pipeline stage
4. News API stores events and streams to connected web clients via SSE
5. Web app displays real-time pipeline status

## Data Model

### Pipeline Events (from News Trader)

```typescript
// === BASE EVENT ===
interface PipelineEvent {
  id: string
  type: EventType
  timestamp: string           // ISO datetime with milliseconds
  newsId: string
}

type EventType =
  | 'news_received'
  | 'news_decision'
  | 'strategy_spawned'
  | 'order_placed'
  | 'order_filled'
  | 'order_cancelled'
  | 'strategy_stopped'

// === NEWS RECEIVED ===
interface NewsReceivedEvent extends PipelineEvent {
  type: 'news_received'
  headline: string
  tickers: string[]
  source: string
  pubTime: string
  newsAge: number             // milliseconds since published
}

// === NEWS DECISION ===
interface NewsDecisionEvent extends PipelineEvent {
  type: 'news_decision'
  decision: 'trade' | 'skip'
  skipReason?: 'no_tickers' | 'no_volume' | 'too_old' | 'market_closed' | 'already_trading'
  volumeCheckMs?: number      // time spent checking volume
  volumeFound?: number        // volume percentage found (if checked)
  volumeThreshold?: number    // threshold used
}

// === STRATEGY SPAWNED ===
interface StrategySpawnedEvent extends PipelineEvent {
  type: 'strategy_spawned'
  strategyId: string
  strategyType: string        // 'volume', 'trend', etc.
  ticker: string
  side: 'long' | 'short'
  positionSizeUsd: number
  exitDelaySeconds: number
}

// === ORDER PLACED ===
interface OrderPlacedEvent extends PipelineEvent {
  type: 'order_placed'
  strategyId: string
  orderId: string
  alpacaOrderId: string
  orderRole: 'entry' | 'exit'
  side: 'buy' | 'sell'
  orderType: 'market' | 'limit'
  qty: number
  limitPrice?: number
}

// === ORDER FILLED ===
interface OrderFilledEvent extends PipelineEvent {
  type: 'order_filled'
  strategyId: string
  orderId: string
  orderRole: 'entry' | 'exit'
  fillPrice: number
  qty: number
  slippage?: number           // vs limit price, if limit order
}

// === ORDER CANCELLED ===
interface OrderCancelledEvent extends PipelineEvent {
  type: 'order_cancelled'
  strategyId: string
  orderId: string
  reason: string
}

// === STRATEGY STOPPED ===
interface StrategyStoppedEvent extends PipelineEvent {
  type: 'strategy_stopped'
  strategyId: string
  reason: 'scheduled_exit' | 'manual' | 'error' | 'cancelled'
  pnl?: number
  pnlPercent?: number
  duration: number            // milliseconds
}
```

### Aggregated Entities

```typescript
// === NEWS EVENT (Aggregated) ===
interface NewsEvent {
  id: string
  headline: string
  tickers: string[]
  source: string
  pubTime: string
  receivedAt: string
  decision: 'trade' | 'skip' | 'pending'
  skipReason?: string
  strategiesSpawned: number
  totalPnl?: number

  // Pipeline timing
  pipeline: {
    newsReceivedAt: string
    decisionAt?: string
    decisionLatencyMs?: number
  }

  // Loaded on demand
  strategies?: Strategy[]
}

// === STRATEGY (Aggregated) ===
interface Strategy {
  id: string
  newsId: string
  ticker: string
  strategyType: string
  side: 'long' | 'short'
  positionSizeUsd: number

  // Status
  status: 'pending_entry' | 'in_position' | 'pending_exit' | 'closed' | 'cancelled'

  // Entry
  entryOrderId?: string
  entryLimitPrice?: number
  entryFillPrice?: number
  entryFilledAt?: string
  entrySlippage?: number
  qty?: number

  // Exit
  exitOrderId?: string
  exitFillPrice?: number
  exitFilledAt?: string
  scheduledExitAt?: string

  // Results
  pnl?: number
  pnlPercent?: number
  duration?: number

  // Timing
  spawnedAt: string
  stoppedAt?: string
  stopReason?: string
}

// === ACTIVE STRATEGY (Live) ===
interface ActiveStrategy extends Strategy {
  status: 'pending_entry' | 'in_position' | 'pending_exit'

  // Live data
  currentPrice?: number
  unrealizedPnl?: number
  unrealizedPnlPercent?: number

  // Countdown
  exitCountdownSeconds?: number
  exitProgress?: number       // 0-100%

  // Entry status (if pending)
  entryPriceDistance?: number // current price vs limit
  entryPriceDistancePercent?: number
  entryWaitingSeconds?: number
}

// === SYSTEM HEALTH ===
interface SystemHealth {
  components: {
    scraper: ComponentHealth
    newsTrader: ComponentHealth
    polygonProxy: ComponentHealth
    alpacaApi: ComponentHealth
    pubsub: ComponentHealth
  }

  latency: {
    newsToDecision: LatencyStats
    decisionToOrder: LatencyStats
    orderToFill: LatencyStats
    totalNewsToFill: LatencyStats
  }

  recentErrors: ErrorEntry[]
}

interface ComponentHealth {
  status: 'healthy' | 'degraded' | 'down' | 'unknown'
  lastSeen?: string
  message?: string
  metrics?: Record<string, number>
}

interface LatencyStats {
  avg: number
  p50: number
  p99: number
  max: number
  sampleCount: number
}

interface ErrorEntry {
  timestamp: string
  component: string
  message: string
  severity: 'warning' | 'error'
}

// === SKIP ANALYSIS ===
interface SkipAnalysis {
  period: string              // 'today', 'week', etc.
  total: number
  traded: number
  skipped: number

  byReason: {
    reason: string
    count: number
    percentage: number
  }[]

  nearMisses: {
    newsId: string
    headline: string
    ticker: string
    reason: string
    volumeFound?: number
    volumeThreshold?: number
    volumeGapPercent?: number
  }[]
}
```

### App State

```typescript
interface AppState {
  // Connection
  sseConnected: boolean
  lastEventAt: string | null

  // Live data
  liveEvents: PipelineEvent[]       // Rolling buffer, last 100
  activeStrategies: ActiveStrategy[]

  // Current selections
  selectedNewsId: string | null
  selectedStrategyId: string | null
  dateRange: DateRange

  // Cached data
  newsHistory: NewsEvent[]
  completedTrades: Strategy[]
  systemHealth: SystemHealth
  skipAnalysis: SkipAnalysis

  // UI state
  activeTab: TabType
  soundEnabled: boolean
  notifications: Notification[]
}

type TabType = 'live' | 'pipeline' | 'active' | 'trades' | 'history' | 'stats' | 'health'
```

### SSE Event Stream

Web app connects to `/stream` and receives events as they happen:

```typescript
// Connect to SSE
const eventSource = new EventSource('http://localhost:8100/stream', {
  headers: { 'X-API-Key': apiKey }
})

// Event types
eventSource.addEventListener('news_received', (e) => {
  const event: NewsReceivedEvent = JSON.parse(e.data)
  // Add to live feed
})

eventSource.addEventListener('news_decision', (e) => {
  const event: NewsDecisionEvent = JSON.parse(e.data)
  // Update news item with decision
})

eventSource.addEventListener('strategy_spawned', (e) => {
  const event: StrategySpawnedEvent = JSON.parse(e.data)
  // Add to active strategies
})

eventSource.addEventListener('order_filled', (e) => {
  const event: OrderFilledEvent = JSON.parse(e.data)
  // Update strategy status
})

eventSource.addEventListener('strategy_stopped', (e) => {
  const event: StrategyStoppedEvent = JSON.parse(e.data)
  // Move from active to completed
})

// Heartbeat every 30s
eventSource.addEventListener('heartbeat', (e) => {
  // Keep connection alive
})
```

## Key Design Decisions

1. **Pipeline-Centric Architecture**: Everything revolves around the news â†’ trade pipeline, not just trade results

2. **Event-Driven Updates**: SSE from News API provides real-time updates without polling

3. **News Trader as Event Source**: All events originate from the trading system, ensuring the web shows exactly what happened

4. **Lightweight Charts**: Used for price charts in pipeline view and stats charts

5. **Sound Notifications**: Optional audio alerts for new tradeable news

6. **Active Strategy Controls**: Manual intervention (early exit, extend timer) when needed

7. **Skip Analysis**: Understanding why news wasn't traded is as important as tracking what was

8. **Latency Monitoring**: Critical for news trading - track timing at each pipeline stage
